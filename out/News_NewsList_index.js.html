const NewsList = props => {
  return (
    <div>
      {
        news
          .filter(item => item.category === props.activeTab)
          .map((item, i) => (
              <div key={i}>
                <img src={item.pic} alt={item.title} />
                <a href='#' >{item.title}</a>
              </div>
            )
          )
      }
    </div>
  );
};

NewsList.propTypes = {
  /** Передает строку с активным разделом новостей */
  activeTab: PropTypes.string.isRequired,
};

export default NewsList;
